nbafinal <- nba %>% left_join(crosswalk, by=c("Team"="Team")) %>% filter(is.na(City) == FALSE) %>%
 arrange(desc(Team))
mlbfinal <- mlb %>% left_join(crosswalk, by=c("Tm"="Team")) %>% filter(is.na(City) == FALSE) %>%
 arrange(desc(Tm))
nflfinal <- nfl %>% left_join(crosswalk, by=c("Tm"="Team")) %>% filter(is.na(City) == FALSE) %>%
 arrange(desc(Tm))
nhlfinal <- nhlfinal %>% separate(Overall, into=c("W", "L", "T"), sep="-") %>% 
  mutate("W-L%" = as.numeric(W)/(as.numeric(W)+as.numeric(L)+as.numeric(T)))
nbafinal <- nbafinal %>% separate(Overall, into=c("W", "L"), sep="-") %>% 
  mutate("W-L%" = as.numeric(W)/(as.numeric(W)+as.numeric(L)))
nbapct <- nbafinal %>% 
  group_by(City) %>%
summarise(totalwins = sum(as.numeric(W)),
          totallosses = sum(as.numeric(L))) %>%
mutate(citywinpercent = totalwins/(totalwins+totallosses), Sport="Basketball") %>%
  select(City, Sport, citywinpercent)
mlbpct <- mlbfinal %>% 
  group_by(City) %>%
summarise(totalwins = sum(as.numeric(W)),
          totallosses = sum(as.numeric(L))) %>%
mutate(citywinpercent = totalwins/(totalwins+totallosses), Sport="Baseball") %>%
  select(City, Sport, citywinpercent)
nflpct <- nflfinal %>% 
  mutate(T = ifelse(is.na(T),0,T)) %>%
  group_by(City) %>%
summarise(totalwins = sum(as.numeric(W)),
          totalties = sum(as.numeric(T)),
          totallosses = sum(as.numeric(L))) %>%
mutate(citywinpercent = totalwins/(totalwins+totalties+totallosses), Sport="Football") %>%
  select(City, Sport, citywinpercent)
nhlpct <- nhlfinal %>%
